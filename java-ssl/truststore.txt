# first, export the public cert from the keystore
keytool -export -alias xxxssl  -keystore /tmp/xxxssl.keystore -file /tmp/xxxssl.cert -storepass password
keytool -exportcert -alias xxxssl  -keystore /tmp/xxxssl.keystore -file /tmp/xxxssl.cert -storepass password

keytool -printcert -file /tmp/xxxssl.cert



# second, import the public cert(generated in the last step) to the truststore
keytool -import -file /tmp/xxx-ssl.cert -alias xxx-ssl -keystore /tmp/xxx-ssl.truststore -storepass password -noprompt
