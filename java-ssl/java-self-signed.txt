# JKS
keytool -genkeypair -alias tls -keyalg RSA -keystore /tmp/tls.jks -keypass password -storepass password -storetype JKS \
-dname "CN=tls,OU=,O=,L=,ST=,C="

# PKCS12, ending with pfx or p12 suffix
keytool -genkeypair -alias tls -keyalg RSA -keystore /tmp/tls.pfx -keypass password -storepass password -storetype PKCS12 \
-dname "CN=tls,OU=,O=,L=,ST=,C="



keytool -list -v -keystore /tmp/tls.jks -storepass password
keytool -list -v -keystore /tmp/tls.pfx -storepass password



# extract public key from jks
keytool -exportcert -alias tls  -keystore /tmp/tls.jks -file /tmp/tls.cert -storepass password

# extra private key
openssl pkcs12 -in /tmp/tls.keystore.p12 -nodes -nocerts -out tls.key

# generate csr
keytool -certreq -alias tls -keystore /tmp/tls.jks -file /tmp/tls.csr -keypass password -storepass password


# root ca
keytool -genkeypair -alias rootca -keyalg RSA -keystore rootca.jks -keypass password -storepass password -storetype PKCS12 \
-dname "CN=rootca,OU=,O=,L=,ST=,C="



# sign with root ca
keytool -gencert -alias rootca -keystore rootca.jks -infile tls.csr -outfile tls.crt -storepass password
